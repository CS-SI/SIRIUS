   template <typename SmartPtr>
   auto MakeSmartPtrArraySpan(const SmartPtr& smart_ptr, const Size& size) {
       return gsl::span<typename SmartPtr::element_type>(smart_ptr.get(),
                                                         size.CellCount());
   }
